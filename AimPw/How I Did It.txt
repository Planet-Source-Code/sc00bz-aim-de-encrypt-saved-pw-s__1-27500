Private Function DeCode(PW As String) As String
    Dim A As Integer, B As Byte, C As Byte, D As Byte
    For A = 1 To Len(PW) Step 2
        C = (Int(A / 2) Mod 16) + 1
        B = Asc(Mid(PW, A, 1)) - 65
'First Letter
        If C = 4 Or C = 9 Or C = 14 Then B = B + IIf(B Mod 2 < 1, 1, -1)
        If C = 5 Or C = 10 Or C = 11 Or C = 15 Or C = 16 Then B = B + IIf(B Mod 4 < 2, 2, -2)
        If C = 1 Or C = 6 Or C = 11 Or C = 12 Then B = B + IIf(B Mod 8 < 4, 4, -4)
        If C = 2 Or C = 7 Or C = 13 Or C = 16 Then B = B + IIf(B < 8, 8, -8)
        If C = 12 Or C = 13 Or C = 14 Or C = 15 Then B = 15 - B
        D = B * 16
        B = Asc(Mid(PW, A + 1, 1)) - 65
'Second Letter
        If C = 5 Or C = 7 Or C = 8 Or C = 9 Or C = 16 Then B = B + IIf(B Mod 2 < 1, 1, -1)
        If C = 1 Or C = 6 Or C = 8 Or C = 9 Or C = 11 Then B = B + IIf(B Mod 4 < 2, 2, -2)
        If C = 2 Or C = 13 Then B = B + IIf(B Mod 8 < 4, 4, -4)
        If C = 3 Or C = 12 Or C = 14 Then B = B + IIf(B < 8, 8, -8)
        If C = 10 Or C = 11 Then B = 15 - B
        DeCode = DeCode & Chr(D + B)
    Next
End Function
